Brute Force
A nested loop approach checking all possible pairs in the array until a match equal to the target is found. This evaluates every combination.
C++ 
vector<int> twoSum(vector<int>& nums, int target) {
    for (int i = 0; i < nums.size(); i++) {
        for (int j = i + 1; j < nums.size(); j++) {
            if (nums[i] + nums[j] == target) {
                return {i, j};
            }
        }
    }
    return {};
}
Optimized
Utilizes a hash map to record elements and their indices. For each element, it checks if the complement exists in the map, turning the lookup into O(1) time.
C++ 
vector<int> twoSum(vector<int>& nums, int target) {
    unordered_map<int, int> map;
    for (int i = 0; i < nums.size(); i++) {
        int complement = target - nums[i];
        if (map.count(complement)) {
            return {map[complement], i};
        }
        map[nums[i]] = i;
    }
    return {};
}
